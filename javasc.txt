var arr=[];
function addNames(){
    var textBox,length,text;
    textBox = document.getElementById("name");
    var name=textBox.value;
    arr.push(name);
    length=arr.length;
    text="[";
    for (var i = 0; i < length; i++) {
        text += arr[i]+",";
    }
    textBox.value="";
    document.getElementById("display").innerHTML =text;
